StructOptimizer Package Repository

Source Code & Documentation

APT (Debian/Ubuntu)

echo "deb [trusted=yes] https://gamelife1314.github.io/structoptimizer/apt stable main" | sudo tee /etc/apt/sources.list.d/structoptimizer.list
sudo apt update && sudo apt install structoptimizer

YUM/DNF (RHEL/Fedora)

sudo tee /etc/yum.repos.d/structoptimizer.repo <<EOF
[structoptimizer]
name=StructOptimizer
baseurl=https://gamelife1314.github.io/structoptimizer/yum
enabled=1
gpgcheck=0
EOF
sudo yum install structoptimizer
